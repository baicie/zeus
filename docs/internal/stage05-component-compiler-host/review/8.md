```ts
// packages/web-c/bundler-plugin/__tests__/componentTransformFilter.spec.ts

import path from 'node:path'
import { describe, expect, it } from 'vitest'
import { createComponentTransformFilter } from '../src/componentTransformFilter'

describe('createComponentTransformFilter', () => {
  it('only transforms files matched by component include', () => {
    const root = path.resolve('/project')

    const shouldTransform = createComponentTransformFilter({
      root,
      include: ['src/components/**/*.{ts,tsx}'],
      exclude: ['src/shared/**', '**/*.test.*'],
    })

    expect(
      shouldTransform(path.resolve(root, 'src/components/button.tsx')),
    ).toBe(true)

    expect(shouldTransform(path.resolve(root, 'src/App.tsx'))).toBe(false)

    expect(shouldTransform(path.resolve(root, 'src/shared/utils.ts'))).toBe(
      false,
    )

    expect(
      shouldTransform(path.resolve(root, 'src/components/button.test.tsx')),
    ).toBe(false)
  })

  it('ignores files outside root', () => {
    const shouldTransform = createComponentTransformFilter({
      root: path.resolve('/project'),
      include: ['src/**/*.{ts,tsx}'],
      exclude: [],
    })

    expect(shouldTransform(path.resolve('/other/src/button.tsx'))).toBe(false)
  })
})
```

## 2. 给 `mergeExternal()` 补单测

现在代码已经修对了，但这个点也容易回归。建议补：

```ts
// packages/web-c/bundler-plugin/__tests__/viteExternal.spec.ts

import { describe, expect, it } from 'vitest'
import { mergeExternal } from '../src/vite'

describe('mergeExternal', () => {
  it('uses plugin external when user external is empty', () => {
    expect(mergeExternal(undefined, ['react', 'vue'])).toEqual(['react', 'vue'])
  })

  it('merges array external', () => {
    expect(mergeExternal(['lodash'], ['react'])).toEqual(['lodash', 'react'])

  it('merges function external', () => {
    const external = mergeExternal(source => source === 'lodash', ['react'])

    expect(typeof external).toBe('function')

    if (typeof external === 'function') {
      expect(external('react', undefined, false)).toBe(true)
      expect(external('lodash', undefined, false)).toBe(true)
      expect(external('vue', undefined, false)).toBe(false)

})
```

## 3. 给 `output-css` 补最小单测

`output-css` 已经是正式发布包，默认会在 `componentLibrary()` 里用。当前实现支持自动查找 `src/styles.css`、watch、生成 asset。

建议至少补：

```txt

没有 CSS 文件时抛出清晰错误
```

测试草案：

```ts
// packages/web-c/output-css/__tests__/output-css.spec.ts

import fs from 'node:fs/promises'
import os from 'node:os'
import path from 'node:path'
import { describe, expect, it, vi } from 'vitest'
import css from '../src'

function createCtx(root: string) {
  return {
    root,
    manifest: {
      version: 1,
      components: [],
    },
    diagnostics: [],
    dts: {
      enabled: true,
      mode: 'auto',
      reason: [],
    },
    outputs: {} as any,
    emitFile: vi.fn(),
    warn: vi.fn(),
    error: vi.fn(),
    addWatchFile: vi.fn(),
    meta: {
      watchMode: false,
    },
  }
}

describe('output-css', () => {
  it('auto detects src/styles.css and emits styles.css', async () => {
    const root = await fs.mkdtemp(path.join(os.tmpdir(), 'zeus-css-'))
    await fs.mkdir(path.join(root, 'src'), { recursive: true })
    await fs.writeFile(
      path.join(root, 'src/styles.css'),
      '.button { color: red; }',
    )

    const plugin = css()
    const ctx = createCtx(root)

    await plugin.buildStart?.(ctx)
    const files = await plugin.generateBundle?.(ctx, {} as any)

    expect(files).toEqual([
      {
        type: 'asset',
        fileName: 'styles.css',
        source: '.button { color: red; }',
      },
    ])

    expect(ctx.addWatchFile).toHaveBeenCalledWith(
      path.resolve(root, 'src/styles.css'),
    )
  })

  it('supports explicit input', async () => {
    const root = await fs.mkdtemp(path.join(os.tmpdir(), 'zeus-css-'))
    await fs.mkdir(path.join(root, 'src'), { recursive: true })
    await fs.writeFile(path.join(root, 'src/theme.css'), ':root {}')

    const plugin = css('src/theme.css')
    const ctx = createCtx(root)

    await plugin.buildStart?.(ctx)
    const files = await plugin.generateBundle?.(ctx, {} as any)

    expect(files?.[0]).toMatchObject({
      type: 'asset',
      fileName: 'styles.css',
      source: ':root {}',
    })
  })
})
```

# C. 可以放后续，不要卡本次合并

## 1. `css()` 没有文件时是否静默跳过

这对 `componentLibrary()` preset 来说有一点偏严格：如果用户没有 CSS 文件但用了 preset，会失败。

可以后续加：

```ts
css({
  optional: true,
})
```

```ts
styles?: string | false | 'auto'
```

但这不是当前必须改的点。现在 examples/headless 有 `src/styles.css`，主流程没问题。
